    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, setDoc, getDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'roteiro-ia-v1';
        const apiKey = ""; 

        window.itineraryData = [];
        let map = null, polyline = null;
        let currentDayId = 1;
        window.selectedTier = 'standard';

        function startSplashAnimation() {
            const logoContainer = document.getElementById('animated-logo');
            const text = "Roteiro.IA";
            logoContainer.innerHTML = '';
            text.split('').forEach((char, i) => {
                const span = document.createElement('span');
                span.innerHTML = char;
                span.className = 'char';
                if (char === '.' || i > 6) span.classList.add('text-emerald-500');
                span.style.animationDelay = `${i * 0.1}s`;
                logoContainer.appendChild(span);
            });
        }
        startSplashAnimation();

        const initAuth = async () => {
            if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                await signInWithCustomToken(auth, __initial_auth_token);
            } else {
                await signInAnonymously(auth);
            }
        };
        initAuth();

        onAuthStateChanged(auth, async (user) => {
            if (user) {
                const tripDoc = await getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'lastTrip'));
                if (tripDoc.exists()) {
                    window.itineraryData = JSON.parse(tripDoc.data().itinerary);
                    showAppUI();
                    loadDay(1);
                }
                setTimeout(() => {
                    document.getElementById('splash').style.opacity = '0';
                    setTimeout(() => document.getElementById('splash').style.display = 'none', 600);
                }, 2000);
            }
        });

        function showAppUI() {
            document.getElementById('input-screen').style.display = 'none';
            document.getElementById('day-selector').classList.remove('hidden');
            document.getElementById('bottom-navigation').classList.remove('hidden');
            document.getElementById('edit-button').classList.remove('hidden');
            document.getElementById('itinerary-view').classList.replace('hide', 'show');
            generateDayPicker();
            if (!map) initMap();
        }

        function initMap() {
            map = L.map('map', { zoomControl: false, attributionControl: false }).setView([0,0], 2);
            L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png').addTo(map);
        }

        window.setTier = (t) => {
            window.selectedTier = t;
            document.querySelectorAll('.choice-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(t === 'económico' ? 't-eco' : t === 'standard' ? 't-std' : 't-pre').classList.add('active');
        };

        window.generateItinerary = async () => {
            const dest = document.getElementById('user-destination').value;
            const start = document.getElementById('trip-start').value;
            const end = document.getElementById('trip-end').value;
            if (!dest || !start || !end) return;

            const btn = document.getElementById('gen-btn');
            btn.innerHTML = '<div class="loader mx-auto"></div>';
            
            const prompt = `Cria um roteiro de viagem para ${dest} de ${start} até ${end} com orçamento ${window.selectedTier}. Responde apenas com um JSON array no seguinte formato: [{"id":1,"title":"Dia 1","subtitle":"Cidade","weather":"22°C","hotel":{"name":"Hotel X","address":"Rua Y"},"total":"€100","timeline":[{"time":"09:00","desc":"Visita ao Museu","price":"€20"}],"coords":[[lat,lng]]}]`;

            try {
                const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], generationConfig: { responseMimeType: "application/json" } })
                });
                const json = await res.json();
                window.itineraryData = JSON.parse(json.candidates[0].content.parts[0].text);
                
                await setDoc(doc(db, 'artifacts', appId, 'users', auth.currentUser.uid, 'settings', 'lastTrip'), { itinerary: JSON.stringify(window.itineraryData) });
                showAppUI();
                loadDay(1);
            } catch (e) {
                btn.innerText = "ERRO AO GERAR";
            } finally {
                btn.innerText = "Gerar Roteiro";
            }
        };

        function generateDayPicker() {
            const container = document.getElementById('day-selector');
            container.innerHTML = '';
            window.itineraryData.forEach(day => {
                const div = document.createElement('div');
                div.className = `day-bubble ${day.id === currentDayId ? 'active' : ''}`;
                div.innerHTML = `<span class="text-2xl font-black">${day.id}</span>`;
                div.onclick = () => loadDay(day.id);
                container.appendChild(div);
            });
        }

        function loadDay(id) {
            currentDayId = id;
            const data = window.itineraryData.find(d => d.id === id);
            if (!data) return;
            document.getElementById('day-title').innerText = data.title;
            document.getElementById('day-subtitle').innerText = data.subtitle;
            document.getElementById('weather-val').innerText = data.weather;
            
            const content = document.getElementById('itinerary-content');
            content.innerHTML = `
                <div class="premium-card border-l-4 border-l-amber-400">
                    <span class="text-[10px] font-black text-neutral-500 uppercase block mb-1">Alojamento</span>
                    <h4 class="text-xl font-extrabold text-white">${data.hotel.name}</h4>
                    <p class="text-xs text-neutral-500">${data.hotel.address}</p>
                </div>
                <div class="premium-card">
                    <span class="text-[10px] font-black text-neutral-500 uppercase block mb-6">Programação</span>
                    <ul class="space-y-6 border-l-2 border-neutral-900 ml-2 pl-8 relative">
                        ${data.timeline.map(item => `
                            <li class="relative mb-4 flex justify-between">
                                <div class="text-sm font-bold text-neutral-200">${item.time} - ${item.desc}</div>
                                <div class="text-[10px] font-black text-emerald-500">${item.price}</div>
                            </li>
                        `).join('')}
                    </ul>
                </div>
            `;
            lucide.createIcons();
        }

        window.openEditor = () => document.getElementById('input-screen').style.display = 'flex';
        window.closeEditor = () => document.getElementById('input-screen').style.display = 'none';
        window.switchView = (v) => {
            document.getElementById('itinerary-view').className = v === 'itinerary' ? 'view-section show' : 'view-section hide';
            document.getElementById('map-view').className = v === 'map' ? 'view-section show' : 'view-section hide';
            document.getElementById('nav-itinerary').className = v === 'itinerary' ? 'nav-btn active' : 'nav-btn';
            document.getElementById('nav-map').className = v === 'map' ? 'nav-btn active' : 'nav-btn';
            if (v === 'map') setTimeout(() => map.invalidateSize(), 100);
        };
    </script>
