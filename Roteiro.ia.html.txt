    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, setDoc, getDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'roteiro-v3-black-green';
        const apiKey = ""; // API Key fornecida pelo ambiente em tempo de execução

        window.itineraryData = [];
        let map = null, polyline = null, markers = [];
        let currentDayId = 1;
        window.selectedTier = 'standard';
        window.selectedAcc = 'hotel';
        window.tripDistances = {};

        function startSplashAnimation() {
            const logoContainer = document.getElementById('animated-logo');
            const text = "Roteiro.IA";
            logoContainer.innerHTML = '';
            text.split('').forEach((char, i) => {
                const span = document.createElement('span');
                span.innerHTML = char === ' ' ? '&nbsp;' : char;
                span.className = 'char';
                if (char === '.' || i > 6) span.classList.add('text-amber-400');
                span.style.animationDelay = `${i * 0.1}s`;
                logoContainer.appendChild(span);
            });
        }
        startSplashAnimation();

        // Autenticação com Retry e suporte a custom token
        const initAuth = async () => {
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (err) {
                console.error("Erro auth:", err);
            }
        };
        initAuth();

        onAuthStateChanged(auth, async (user) => {
            if (user) {
                const tripDoc = await getDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'settings', 'lastTrip'));
                if (tripDoc.exists()) {
                    window.itineraryData = JSON.parse(tripDoc.data().itinerary);
                    showAppUI();
                    loadDay(1);
                }
                setTimeout(() => {
                    document.getElementById('splash').style.opacity = '0';
                    setTimeout(() => document.getElementById('splash').style.display = 'none', 600);
                }, 2500);
            }
        });

        function showAppUI() {
            document.getElementById('input-screen').style.display = 'none';
            document.getElementById('day-selector').classList.remove('hidden');
            document.getElementById('bottom-navigation').classList.remove('hidden');
            document.getElementById('edit-button').classList.remove('hidden');
            document.getElementById('itinerary-view').classList.replace('hide', 'show');
            generateDayPicker();
            calculateGrandTotal();
            if (!map) initMap();
        }

        function initMap() {
            map = L.map('map', { zoomControl: false, attributionControl: false }).setView([0,0], 2);
            L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png').addTo(map);
        }

        window.setTier = (t) => {
            window.selectedTier = t;
            document.querySelectorAll('.choice-btn').forEach(b => b.classList.remove('active'));
            const mapId = { 'econômico': 't-eco', 'standard': 't-std', 'premium': 't-pre' };
            document.getElementById(mapId[t]).classList.add('active');
        };

        window.setAcc = (a) => {
            window.selectedAcc = a;
            document.querySelectorAll('.acc-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(`a-${a}`).classList.add('active');
        };

        // Função de geração com tratamento de erro e retentativas (Exponential Backoff)
        window.generateItinerary = async () => {
            const origin = document.getElementById('user-origin').value;
            const destination = document.getElementById('user-destination').value;
            const startDate = document.getElementById('trip-start').value;
            const endDate = document.getElementById('trip-end').value;
            const extraQuery = document.getElementById('user-query').value;
            
            if (!destination || !startDate || !endDate) {
                const msg = document.createElement('div');
                msg.className = "fixed top-10 left-1/2 -translate-x-1/2 bg-red-600 text-white px-6 py-3 rounded-xl z-[3000] font-bold";
                msg.innerText = "Preencha o destino e as datas.";
                document.body.appendChild(msg);
                setTimeout(() => msg.remove(), 3000);
                return;
            }
            
            const btn = document.getElementById('gen-btn');
            btn.innerHTML = '<div class="loader mx-auto"></div>';
            btn.disabled = true;

            const tripContext = `Viagem de ${origin || 'origem não especificada'} para ${destination}. Período: ${startDate} a ${endDate}. Orçamento: ${window.selectedTier}. Acomodação: ${window.selectedAcc}. Notas extras: ${extraQuery}`;
            const prompt = `Gere um roteiro detalhado para: ${tripContext}. JSON format: [{ "id": 1, "title": "Dia 1: ...", "subtitle": "Local", "weather": "25°C", "hotel": {"name": "X", "address": "Y"}, "ticket": "R$ X", "meal": "R$ X", "total": "R$ X", "timeline": [{ "time": "08:00", "desc": "Atividade", "price": "R$ X" }], "coords": [[lat,lng], [lat,lng]] }]`;

            const callGemini = async (retries = 0) => {
                try {
                    const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ 
                            contents: [{ parts: [{ text: prompt }] }], 
                            generationConfig: { responseMimeType: "application/json" } 
                        })
                    });
                    
                    if (!res.ok) throw new Error("API Failure");
                    
                    const json = await res.json();
                    const resultText = json.candidates?.[0]?.content?.parts?.[0]?.text;
                    if (!resultText) throw new Error("Empty Response");
                    
                    window.itineraryData = JSON.parse(resultText);
                    
                    if (auth.currentUser) {
                        await setDoc(doc(db, 'artifacts', appId, 'users', auth.currentUser.uid, 'settings', 'lastTrip'), { 
                            itinerary: JSON.stringify(window.itineraryData) 
                        });
                    }
                    
                    showAppUI();
                    loadDay(1);
                } catch (e) {
                    if (retries < 5) {
                        const delay = Math.pow(2, retries) * 1000;
                        setTimeout(() => callGemini(retries + 1), delay);
                    } else {
                        btn.innerText = "ERRO NA IA";
                        btn.disabled = false;
                    }
                }
            };

            callGemini();
        };

        function generateDayPicker() {
            const container = document.getElementById('day-selector');
            container.innerHTML = '';
            window.itineraryData.forEach(day => {
                const div = document.createElement('div');
                div.className = `day-bubble ${day.id === currentDayId ? 'active' : ''}`;
                div.id = `bubble-${day.id}`;
                div.innerHTML = `<span class="text-[8px] uppercase font-black opacity-40">D</span><span class="text-2xl font-black">${day.id}</span>`;
                div.onclick = () => loadDay(day.id);
                container.appendChild(div);
            });
        }

        function loadDay(id) {
            currentDayId = id;
            const data = window.itineraryData.find(d => d.id === id);
            if (!data) return;
            document.querySelectorAll('.day-bubble').forEach(b => b.classList.remove('active'));
            document.getElementById(`bubble-${id}`)?.classList.add('active');
            document.getElementById('day-title').innerText = data.title;
            document.getElementById('day-subtitle').innerText = data.subtitle;
            document.getElementById('weather-val').innerText = data.weather.split(' ')[0];
            document.getElementById('hotel-name').innerText = data.hotel.name;
            document.getElementById('hotel-address').innerText = data.hotel.address;
            document.getElementById('ticket-val').innerText = data.ticket;
            document.getElementById('meal-val').innerText = data.meal;
            document.getElementById('total-val').innerText = data.total;
            document.getElementById('map-day-title').innerText = data.title;
            const tl = document.getElementById('timeline');
            tl.innerHTML = '';
            data.timeline.forEach(item => {
                const li = document.createElement('li');
                li.className = "relative mb-4 flex justify-between items-start";
                li.innerHTML = `
                    <div class="flex items-start">
                        <div class="absolute -left-[37px] w-4 h-4 bg-amber-400 rounded-full border-4 border-black"></div>
                        <div class="text-sm font-bold text-neutral-200 leading-tight">${item.time} - ${item.desc}</div>
                    </div>
                    <div class="text-[10px] font-black text-emerald-500 ml-4">${item.price}</div>`;
                tl.appendChild(li);
            });
            document.getElementById('grand-total-section').classList.toggle('hidden', id !== 1);
            if (map) updateMapRoute(data);
            lucide.createIcons();
        }

        async function updateMapRoute(data) {
            if (!map || !data.coords || data.coords.length < 1) return;
            if (polyline) map.removeLayer(polyline);
            markers.forEach(m => map.removeLayer(m));
            markers = [];
            const points = data.coords.map(c => `${c[1]},${c[0]}`).join(';');
            const osrmUrl = `https://router.project-osrm.org/route/v1/driving/${points}?overview=full&geometries=geojson`;
            try {
                const res = await fetch(osrmUrl);
                const json = await res.json();
                if (json.code === 'Ok' && json.routes?.[0]) {
                    const streetGeometry = json.routes[0].geometry.coordinates.map(c => [c[1], c[0]]);
                    window.tripDistances[data.id] = json.routes[0].distance / 1000;
                    polyline = L.polyline(streetGeometry, { color: '#fbbf24', weight: 6, opacity: 0.8 }).addTo(map);
                    document.getElementById('map-day-km').innerText = (json.routes[0].distance / 1000).toFixed(1);
                    document.getElementById('map-km-label').innerText = `Rota: ${(json.routes[0].distance / 1000).toFixed(1)}km`;
                }
            } catch (e) {
                polyline = L.polyline(data.coords, { color: '#fbbf24', weight: 4 }).addTo(map);
            }
            data.coords.forEach((c, idx) => {
                markers.push(L.circleMarker(c, { radius: 6, color: '#000', fillColor: idx === 0 ? '#10b981' : '#fbbf24', fillOpacity: 1, weight: 2 }).addTo(map));
            });
            if (polyline) map.fitBounds(polyline.getBounds(), { padding: [50, 50] });
            updateTotalKmDisplay();
        }

        function updateTotalKmDisplay() {
            let total = 0;
            Object.values(window.tripDistances).forEach(val => total += val);
            document.getElementById('map-total-trip-km').innerText = `${total.toFixed(1)} KM`;
        }

        window.switchView = (v) => {
            const btnItin = document.getElementById('nav-itinerary'), btnMap = document.getElementById('nav-map');
            if (v === 'map') {
                document.getElementById('itinerary-view').classList.replace('show', 'hide');
                document.getElementById('map-view').classList.replace('hide', 'show');
                btnMap.classList.add('active'); btnItin.classList.remove('active');
                setTimeout(() => { map.invalidateSize(); const currentData = window.itineraryData.find(d => d.id === currentDayId); if (currentData) updateMapRoute(currentData); }, 300);
            } else {
                document.getElementById('map-view').classList.replace('show', 'hide');
                document.getElementById('itinerary-view').classList.replace('hide', 'show');
                btnItin.classList.add('active'); btnMap.classList.remove('active');
            }
        };

        function calculateGrandTotal() {
            let total = 0;
            window.itineraryData.forEach(d => {
                const n = parseFloat(d.total.replace(/[^\d,]/g, '').replace(',', '.'));
                if (!isNaN(n)) total += n;
            });
            document.getElementById('grand-total-val').innerText = `R$ ${total.toLocaleString('pt-BR')}`;
        }

        window.closeEditor = () => document.getElementById('input-screen').style.display = 'none';
